def cov_matrix(data, verbose=False):
    covariance_matrix = np.cov(data, rowvar=False)
    if is_pos_def(covariance_matrix):
        inv_covariance_matrix = np.linalg.inv(covariance_matrix)
        if is_pos_def(inv_covariance_matrix):
            return covariance_matrix, inv_covariance_matrix
        else:
            print("Error: Inverse of Covariance Matrix is not positive definite!")
    else:
        print("Error: Covariance Matrix is not positive definite!")

def MahalanobisDist(inv_cov_matrix, mean_distr, data, verbose=False):
    inv_covariance_matrix = inv_cov_matrix
    vars_mean = mean_distr
    diff = data - vars_mean
    md = []
    for i in range(len(diff)):
        md.append(np.sqrt(diff[i].dot(inv_covariance_matrix).dot(diff[i])))
    return md

Threshold value to detect anomalies, it has two options: regular or conservative¶

In [14]:
def MD_threshold(dist, conservative=False):
    k = 3. if conservative else 2.
    threshold = np.mean(dist) * k
    return threshold

Define matrix whether positive definite or not¶

In [15]:
def is_pos_def(A):
    if np.allclose(A, A.T):
        try:
            np.linalg.cholesky(A)
            return True
        except np.linalg.LinAlgError:
            return False
    else:
        return False

Mahalanobis distance is the multivariate generalization of finding how many standard deviations away a point is from the mean of the multivariate distribution. Also, the square of the Mahalanobis distance to the centroid of the distribution should follow a Chi-square distribution. Otherwise, it is needed to visualize the Mahalanobis distance's distribution to select a threshold value¶

In order to measure a good balance between precision and recall, it is calculated the F1 score.¶

In [24]:
TP = df_confusion[True][True]
FP = df_confusion[True][False]
TN = df_confusion[False][False]
FN = df_confusion[False][True]

confusion_matrix_accuracy = (TP + TN)/(TP + TN + FP + FN)
confusion_matrix_precision = TP/(TP + FP)
confusion_matrix_recall = TP/(TP + FN)

f1_score = 2*confusion_matrix_precision*confusion_matrix_recall/(confusion_matrix_precision + confusion_matrix_recall)
